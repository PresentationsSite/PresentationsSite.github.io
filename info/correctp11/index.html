Substitution :

def substitution(M) :
    n = len(M)
    V = [0]*n
    for i in range(n-1,-1,-1):
        V[i] = M[i][n]
        for j in range(i+1,n):
            V[i] -= M[i][j]*V[j]
        V[i] /= M[i][i]
    return V

Pivot partiel (recherche de l’indice
d’une valeur max dans une liste) :

def recherchePivotPartiel(M :list,h :int,k :int) -> int:
    ipivot, pivot = h, M[h][k]
    for i in range(h+1,len(M)) :
        if abs(M[i][k]) > pivot :
            pivot = abs(M[i][k])
            ipivot = i
    return ipivot

Interpolation (traduction python
de la formule mathématique)

$$ P(x) = \sum_{j=1}^n y_j\left(\color{purple}\prod_{i=1,i≠j}^n \frac{x-x_i}{x_j-x_i}\color{black}\right) $$

def interp(f, n: int, interv: tuple, x: float) -> float:
    debut,fin = interv[0],interv[1]
    X = [debut + i*(fin-debut)/(n-1) for i in range(n)]
    Y = [f(x) for x in X]
    s = 0
    for i in range(n) :
        p = 1
        for j in range(n) :
            if j != i :
                p *= (x-X[j])/(X[i]-X[j])
        s += Y[i]*p
    return s

def interp_quad(f,interv,N,x) :
    debut,fin = interv[0],interv[1]
    X = [debut + i*(fin-debut)/(N-1) for i in range(N)]
    for i in range(0,N-2,2) :
        if X[i] <= x < X[i+2] :
            return interp(f,3,(X[i],X[i+2]),x)

def Euler(f,t0,tf,y0,h) :
    n = int((tf-t0)/h)
    T = [t0+i*h for i in range(n)]`
    Y = [y0]
    for i in range(1,n) :

s,r,b = 3,26.5,1
x0,y0,z0 = 0,1,1.05
t = 0
h = 1e-2
x,y,z = x0,y0,z0
X,Y,Z = [x0],[y0],[z0]

while t <= 100 :
    x_old, y_old, z_old = x,y,z
    x += (s*(y_old-x_old))*h
    y += (r*x_old - y_old - x_old*z_old)*h
    z += (x_old*y_old - b*z_old)*h
    X.append(x)
    Y.append(y)
