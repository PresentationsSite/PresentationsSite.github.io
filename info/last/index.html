def KNN_interp(x: float, k: int, Pts: list) -> float:
    """
    pr√©conditions : Pts est une liste de tuples contenant les coordonn√©es (x,y) des points d'entra√Ænement
    postcondition : la fonction retourne la moyenne des ordonn√©es des k points dont l'abscisse est la plus proche de x
    """
    Nv_Pts = []
    for pt in Pts :
        Nv_Pts.append((abs(pt[0]-x),pt[1]))
    y = 0
    Nv_Pts.sort()
    for pt in Nv_Pts[:k] :
        y += pt[1]
    y /= k
    return y

def dist(x: list, Pts: list) -> list:
    """
    pr√©conditions : x contient les coordonn√©es d'un point
                    Pts est une liste de tuples ou chaque tuple contient les coordonn√©es d'un point
                    tous les points doivent avoir le m√™me nombre de coordonn√©es
    postcondition : la fonction doit retourner la liste des distances entre le points x et chacun des points de la liste Pts.
    """
    n = len(x)
    for pt in Pts :
        assert n == len(pt), "Les points n'ont pas tous le m√™me nombre de coordonn√©es !"
    L = []
    for pt in Pts :
        d = 0
        for i in range(n) :
            d += (pt[i]-x[i])**2
        d = d**0.5
        L.append(d)
    return L

def attribution(k: int, data: list, Dist: list) -> list:
    """
    pr√©conditions :  Dist est une liste √† deux dimensions contenant k sous-listes
                     chaque sous-liste contient les distances de chaque point de data √† un des centres
                     Une sous-liste correspond √† un centre
    postconditions : la fonction doit retourner (sous forme de liste) le num√©ro de cluster (de 0 √† k-1) associ√© √† chaque point de data
                     Si le premier point appartient au cluster 1, le deuxi√®me au cluster 3, le trois√®me au cluster 0,
                     la liste Clusters commencera ainsi [1,3,0,...]
    """
    Cluster = []
    n = len(data)
    for m in range(n) :
        imin = 0
        mini = Dist[0][m]
        for i in range(1,k) :
            if Dist[i][m] < mini :
                mini = Dist[i][m]
                imin = i
        Cluster.append(imin)
    return Cluster

from collections import deque

def testBipartite(G,depart) :
    """
    precondition :   le graphe G doit √™tre non orient√©
    postconditions : le dictionnaire Vus dit si un sommet a √©t√© explor√© ou non
                     le dictionnaire Distance donne la distance de chaque sommet ajout√© par rapport au sommet de d√©part
                     la fonction retourne Vrai si on tombe sur un sommet d√©j√† explor√© 
                     dont la distance au sommet de d√©part est la m√™me que celle de son pr√©d√©cesseur
                     et Faux si aucun tel cas n'est rencontr√©.
    """
    Vus = {s : False for s in G}
    Distance = {s : 0 for s in G}
    file = deque()
    file.append(depart)
    while file :
        sommet = file.popleft()
        if not Vus[sommet] :
            Vus[sommet] = True
            for s in G[sommet] :
                if Vus[s] :
                    if Distance[s] == Distance[sommet] :
                        return False
                Distance[s] = Distance[sommet] + 1
                file.append(s)
    return True

def inverseGraphe(G: dict) -> dict:
    """
    pr√©condition : G est la liste d'adjacence (sous forme de dictionnaire) d'un graphe orient√©
    """
    inv = {s:[] for s in G}
    for pred in G :
        for succ in G[pred] :
            inv[succ].append(pred)
    return inv
