def cout_tot_min(mat_couts):
  N = len(mat_couts)
  K = len(mat_couts[0])
  M_tot = [[0]*K for i in range(N)]
  for k in range(K):
    M_tot[0][k] = mat_couts[0][k]
  for n in range(1,N):
    for k in range(K):
      M_tot[n][k] = mat_couts[n][k] + min([M_tot[n-1][kprime] for kprime in range(K) if kprime != k])
  return min(M_tot[n-1])

  N = len(M_tot)
